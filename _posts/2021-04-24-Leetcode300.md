```cpp
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int num_size = nums.size();
        if (num_size == 0) {
            return 0;
        }
        
        int kUnusedValue = INT_MAX;
        
        for (int i = 0; i < num_size; i++) {
            int cur_val = nums[i];
            nums[i] = kUnusedValue;
            
            for (int j = 0; j <= i; j++) {
                if (nums[j] == kUnusedValue || cur_val <= nums[j]) {
                    nums[j] = cur_val;
                    break;
                }
            }
            
        }
        
        for (int i = num_size - 1; i >= 0; i--) {
            if (nums[i] != kUnusedValue) {
                return i + 1;
            }
        }
        
        return 0;
    }
};
```
